26class StackEmptyError(Exception): 

27 def __init__(self, message="Stack is empty"): 

28 self.message = message 

29 super().__init__(self.message)

33class ArrayStack: 

34 """LIFO Stack implementation using a Python list as underlying storage.""" 

35 

36 def __init__(self): 

37 """Create an empty stack.""" 

38 self._data = [] # nonpublic list instance 

39 

40 def __len__(self): 

41 """Return the number of elements in the stack.""" 

42 return len(self._data) 

43 

44 def is_empty(self): 

45 """Return True if the stack is empty.""" 

46 return len(self._data) == 0 

47 

48 def push(self, e): 

49 """Add element e to the top of the stack.""" 

50 self._data.append(e) # new item stored at end of list 

51 

52 def top(self): 

53 """Return (but do not remove) the element at the top of the stack. 

54 

55 Raise Empty exception if the stack is empty. 

56 """ 

57 if self.is_empty(): 

58 raise Empty('Stack is empty') 

59 return self._data[-1] # the last item in the list 

60 

61 def pop(self): 

62 """Remove and return the element from the top of the stack (i.e., LIFO). 

63 

64 Raise Empty exception if the stack is empty. 

65 """ 

66 if self.is_empty(): 

67 raise StackEmptyError("Cannot pop from an empty stack") 

68 #raise Empty('Stack is empty') 

69 return self._data.pop() # remove last item from list
